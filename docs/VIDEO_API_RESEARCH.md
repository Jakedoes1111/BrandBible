## üìù Code Example: Creatomate Integration

```typescript
// services/videoGenerator.ts
import axios from 'axios';

const CREATOMATE_API_KEY = import.meta.env.VITE_CREATOMATE_API_KEY;

interface VideoRequest {
  templateId: string;
  modifications: {
    text?: string;
    images?: string[];
    colors?: string[];
    audio?: string;
  };
}

export async function generateVideo(request: VideoRequest): Promise<string> {
  const response = await axios.post(
    'https://api.creatomate.com/v1/renders',
    {
      template_id: request.templateId,
      modifications: request.modifications
    },
    {
      headers: {
        'Authorization': `Bearer ${CREATOMATE_API_KEY}`,
        'Content-Type': 'application/json'
      }
    }
  );

  // Poll for completion
  const renderId = response.data.id;
  let videoUrl = null;

  while (!videoUrl) {
    await new Promise(resolve => setTimeout(resolve, 2000));
    
    const statusResponse = await axios.get(
      `https://api.creatomate.com/v1/renders/${renderId}`,
      {
        headers: { 'Authorization': `Bearer ${CREATOMATE_API_KEY}` }
      }
    );

    if (statusResponse.data.status === 'succeeded') {
      videoUrl = statusResponse.data.url;
    }
  }

  return videoUrl;
}
```
